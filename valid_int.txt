int		ft_isspace(int c)
{
	c = (unsigned char)c;
	if (c == ' ' || c == '\t' || c == '\n' || c == '\v' ||
			c == '\f' || c == '\r')
		return (1);
	return (0);
}

int		ft_atoi_mod(const char *str)
{
	long long	result;
	long long	sign;

	sign = 1;
	result = 0;
	if (str)
	{
		while (ft_isspace(*str))
			str++;
		if (*str == '-')
			sign = -1;
		if (*str == '+' || *str == '-')
			str++;
		while (ft_isdigit(*str))
		{
			result = 10 * result + (*str - '0');
			str++;
		}
	}
	if (result > FT_INT_MAX || result < FT_INT_MIN)
	{
		ft_putstr_fd("Error\n", 2);
		exit(0);
	}
	return ((int)result * sign);
}

int valid_int(char **args)
{
	int i;
	int tmp;
	
	i = 0;
	while (args[i])
	{
		tmp = ft_atoi_mod(args[i]);
		if (tmp > FT_INT_MAX || tmp < FT_INT_MIN)
			return (-1);
	}
		i++;
	return (1);
}
